# рҹҸ—пёҸ РңРҫРҙСғР»СҢРҪР°СҸ Р°СҖС…РёСӮРөРәСӮСғСҖР°

## рҹ“Ӣ РһРұР·РҫСҖ Р°СҖС…РёСӮРөРәСӮСғСҖСӢ

РЎРёСҒСӮРөРјР° РҝРҫСҒСӮСҖРҫРөРҪР° РҪР° РҝСҖРёРҪСҶРёРҝР°С… РјРҫРҙСғР»СҢРҪРҫРіРҫ РјРҫРҪРҫР»РёСӮР° СҒ РёСҒРҝРҫР»СҢР·РҫРІР°РҪРёРөРј Domain-Driven Design (DDD) РҙР»СҸ РҫРұРөСҒРҝРөСҮРөРҪРёСҸ РјР°СҒСҲСӮР°РұРёСҖСғРөРјРҫСҒСӮРё, РҝРҫРҙРҙРөСҖР¶РәРё Рё РІРҫР·РјРҫР¶РҪРҫСҒСӮРё РҝР»Р°РІРҪРҫРіРҫ РҝРөСҖРөС…РҫРҙР° Рә РјРёРәСҖРҫСҒРөСҖРІРёСҒРҪРҫР№ Р°СҖС…РёСӮРөРәСӮСғСҖРө.

## рҹҺҜ РҹСҖРёРҪСҶРёРҝСӢ Р°СҖС…РёСӮРөРәСӮСғСҖСӢ

### 1. **Domain-Driven Design (DDD)**
- Р§РөСӮРәРҫРө СҖР°Р·РҙРөР»РөРҪРёРө РҙРҫРјРөРҪРҫРІ Рё СҒСғРұРҙРҫРјРөРҪРҫРІ
- Bounded Contexts РҙР»СҸ РәР°Р¶РҙРҫРіРҫ РјРҫРҙСғР»СҸ
- Ubiquitous Language РІ СҖР°РјРәР°С… РәР°Р¶РҙРҫРіРҫ РәРҫРҪСӮРөРәСҒСӮР°
- Р”РҫРјРөРҪРҪСӢРө СҒРөСҖРІРёСҒСӢ, СҒСғСүРҪРҫСҒСӮРё Рё value objects

### 2. **Clean Architecture**
- Р—Р°РІРёСҒРёРјРҫСҒСӮРё РҪР°РҝСҖР°РІР»РөРҪСӢ РҫСӮ РІРҪРөСҲРҪРёС… СҒР»РҫРөРІ Рә РІРҪСғСӮСҖРөРҪРҪРёРј
- Р‘РёР·РҪРөСҒ-Р»РҫРіРёРәР° РҪРөР·Р°РІРёСҒРёРјР° РҫСӮ С„СҖРөР№РјРІРҫСҖРәРҫРІ Рё Р‘Р”
- РҳРҪРІРөСҖСҒРёСҸ Р·Р°РІРёСҒРёРјРҫСҒСӮРөР№ СҮРөСҖРөР· РёРҪСӮРөСҖС„РөР№СҒСӢ
- РўРөСҒСӮРёСҖСғРөРјРҫСҒСӮСҢ РҪР° РІСҒРөС… СғСҖРҫРІРҪСҸС…

### 3. **РңРҫРҙСғР»СҢРҪРҫСҒСӮСҢ**
- РЎР»Р°РұР°СҸ СҒРІСҸР·Р°РҪРҪРҫСҒСӮСҢ РјРөР¶РҙСғ РјРҫРҙСғР»СҸРјРё
- Р’СӢСҒРҫРәР°СҸ СҒРІСҸР·РҪРҫСҒСӮСҢ РІРҪСғСӮСҖРё РјРҫРҙСғР»РөР№
- Р§РөСӮРәРёРө РёРҪСӮРөСҖС„РөР№СҒСӢ РјРөР¶РҙСғ РјРҫРҙСғР»СҸРјРё
- РқРөР·Р°РІРёСҒРёРјРҫРө СҖР°Р·РІРёСӮРёРө Рё СӮРөСҒСӮРёСҖРҫРІР°РҪРёРө РјРҫРҙСғР»РөР№

```go
// РҹСҖРёРјРөСҖ РҙРҫРјРөРҪРҪРҫР№ СҒСғСүРҪРҫСҒСӮРё
type Document struct {

    updatedAt   time.Time

}

func (d *Document) Approve(approverID UserID) error {
    if d.status != DocumentStatusPending {
        return ErrDocumentNotPending
    }

    d.status = DocumentStatusApproved
    d.updatedAt = time.Now()

    // РҹСғРұР»РёРәСғРөРј РҙРҫРјРөРҪРҪРҫРө СҒРҫРұСӢСӮРёРө

        DocumentID: d.id,
        ApproverID: approverID,
        ApprovedAt: d.updatedAt,
    })

    return nil
}
```

### 2. **Application Layer (РЎР»РҫР№ РҝСҖРёР»РҫР¶РөРҪРёСҸ)**
```go
// Use Case РҝСҖРёРјРөСҖ
type ApproveDocumentUseCase struct {
    documentRepo DocumentRepository
    userRepo     UserRepository
    eventBus     EventBus
    unitOfWork   UnitOfWork
}

func (uc *ApproveDocumentUseCase) Execute(cmd ApproveDocumentCommand) error {
    return uc.unitOfWork.Execute(func() error {
        // РҹРҫР»СғСҮР°РөРј РҙРҫРәСғРјРөРҪСӮ
        document, err := uc.documentRepo.GetByID(cmd.DocumentID)
        if err != nil {
            return err
        }

        // РҹСҖРҫРІРөСҖСҸРөРј РҝСҖР°РІР° РҝРҫР»СҢР·РҫРІР°СӮРөР»СҸ
        user, err := uc.userRepo.GetByID(cmd.ApproverID)
        if err != nil {
            return err
        }

        if !user.CanApprove(document) {
            return ErrInsufficientPermissions
        }

        // Р’СӢРҝРҫР»РҪСҸРөРј РұРёР·РҪРөСҒ-РҫРҝРөСҖР°СҶРёСҺ
        if err := document.Approve(cmd.ApproverID); err != nil {
            return err
        }

        // РЎРҫС…СҖР°РҪСҸРөРј РёР·РјРөРҪРөРҪРёСҸ
        if err := uc.documentRepo.Save(document); err != nil {
            return err
        }

        // РҹСғРұР»РёРәСғРөРј СҒРҫРұСӢСӮРёСҸ
        for _, event := range document.GetDomainEvents() {
            if err := uc.eventBus.Publish(event); err != nil {
                return err
            }
        }

        return nil
    })
}
```

```go
type PostgresDocumentRepository struct {
    db *sql.DB
}

func (r *PostgresDocumentRepository) GetByID(id DocumentID) (*Document, error) {
    query := `
        SELECT id, title, content, author_id, status, created_at, updated_at
        FROM documents
        WHERE id = $1
    `

    var doc Document
    err := r.db.QueryRow(query, id.String()).Scan(
        &doc.id,
        &doc.title,
        &doc.content,
        &doc.authorID,
        &doc.status,
        &doc.createdAt,
        &doc.updatedAt,
    )

    if err != nil {
        if err == sql.ErrNoRows {
            return nil, ErrDocumentNotFound
        }
        return nil, err
    }

    return &doc, nil
}
```

```go
func SecurityMiddleware(next http.Handler) http.Handler {
    return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
        // Rate limiting
        if !rateLimiter.Allow() {
            http.Error(w, "Rate limit exceeded", http.StatusTooManyRequests)
            return
        }

        // Input validation
        if err := validateRequest(r); err != nil {
            http.Error(w, "Invalid input", http.StatusBadRequest)
            return
        }

        // Security headers
        w.Header().Set("X-Content-Type-Options", "nosniff")
        w.Header().Set("X-Frame-Options", "DENY")
        w.Header().Set("X-XSS-Protection", "1; mode=block")

        next.ServeHTTP(w, r)
    })
}
```
